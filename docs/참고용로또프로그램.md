# 로또 패턴 분석 시스템 (LOTTO Pattern Analysis System)

## 📊 프로젝트 개요

이 프로젝트는 로또 당첨 번호의 다양한 패턴을 분석하여 통계적 인사이트를 제공하는 고성능 분석 시스템입니다. 1167회차의 로또 데이터를 기반으로 16가지 주요 패턴을 분석하여 로또 번호의 출현 특성을 파악합니다.

## 🎯 주요 기능

### 📈 16가지 패턴 분석
1. **번호 일치 패턴** - 이전 회차와의 번호 일치도 분석
2. **홀짝 분포** - 홀수/짝수 번호의 분포 패턴
3. **연속 번호 패턴** - 연속된 번호의 출현 빈도
4. **번호 합계 분포** - 당첨 번호 합계의 구간별 분포
5. **고정 간격 패턴** - 일정한 간격으로 나타나는 번호 패턴
6. **끝자리 패턴** - 번호 끝자리의 동일성 분석
7. **최대 간격 패턴** - 가장 큰 번호와 가장 작은 번호의 차이
8. **구간별 번호 분포** - 1-15, 16-30, 31-45 구간별 분석
9. **번호 평균값 분포** - 당첨 번호 평균의 분포
10. **배수 패턴** - 2, 3, 4, 5의 배수 분포
11. **10구간 분석** - 1-10, 11-20, 21-30, 31-40, 41-45 구간별 세밀 분석
12. **등차수열 패턴** - 등차수열을 이루는 번호 조합
13. **등비수열 패턴** - 등비수열을 이루는 번호 조합
14. **홀짝 교차 패턴** - 홀수와 짝수의 교대 출현
15. **합계 배수 패턴** - 당첨 번호 합계가 특정 수의 배수인 경우
16. **분산도 패턴** - 번호들의 분산 정도 분석

### ⚡ 고성능 필터 최적화
- **병렬 처리**: 멀티프로세싱을 통한 대용량 데이터 처리
- **청크 기반 처리**: 메모리 효율적인 배치 처리
- **NumPy 최적화**: 고속 수치 연산

#### 1. 번호 일치 패턴 분석
```python
def analyze_match_pattern(current_numbers, previous_numbers):
    """이전 회차와의 번호 일치도 계산"""
    current_set = set(current_numbers)
    previous_set = set(previous_numbers)
    match_count = len(current_set.intersection(previous_set))
    return match_count

# 알고리즘: 집합 교집합 연산 O(n)
# 시간복잡도: O(6) = O(1) (로또 번호 6개 고정)
```

#### 2. 홀짝 분포 분석
```python
def analyze_odd_even_pattern(numbers):
    """홀짝 분포 패턴 분석"""
    odd_count = sum(1 for num in numbers if num % 2 == 1)
    even_count = len(numbers) - odd_count
    return {'odd': odd_count, 'even': even_count}

# 알고리즘: 모듈로 연산을 통한 홀짝 판별
# 시간복잡도: O(n), 공간복잡도: O(1)
```

#### 3. 연속 번호 패턴 분석
```python
def analyze_consecutive_pattern(numbers):
    """연속 번호 패턴 분석"""
    sorted_nums = sorted(numbers)
    consecutive_groups = []
    current_group = [sorted_nums[0]]
    
    for i in range(1, len(sorted_nums)):
        if sorted_nums[i] == sorted_nums[i-1] + 1:
            current_group.append(sorted_nums[i])
        else:
            if len(current_group) >= 2:
                consecutive_groups.append(current_group)
            current_group = [sorted_nums[i]]
    
    if len(current_group) >= 2:
        consecutive_groups.append(current_group)
    
    return consecutive_groups

# 알고리즘: 정렬 후 순차 탐색
# 시간복잡도: O(n log n), 공간복잡도: O(n)
```

#### 4. 고정 간격 패턴 분석
```python
def analyze_fixed_interval_pattern(numbers, interval):
    """고정 간격 패턴 분석"""
    sorted_nums = sorted(numbers)
    pattern_count = 0
    
    for i in range(len(sorted_nums) - 1):
        consecutive_count = 1
        current_num = sorted_nums[i]
        
        for j in range(i + 1, len(sorted_nums)):
            expected_num = current_num + interval * consecutive_count
            if sorted_nums[j] == expected_num:
                consecutive_count += 1
                current_num = sorted_nums[j]
            else:
                break
        
        if consecutive_count >= 3:  # 3개 이상 연속
            pattern_count = max(pattern_count, consecutive_count)
    
    return pattern_count

# 알고리즘: 이중 반복문을 통한 패턴 탐지
# 시간복잡도: O(n²), 공간복잡도: O(1)
```

#### 5. 등차수열 패턴 분석
```python
def analyze_arithmetic_sequence(numbers):
    """등차수열 패턴 분석"""
    if len(numbers) < 3:
        return 0
    
    sorted_nums = sorted(numbers)
    max_sequence = 0
    
    for i in range(len(sorted_nums) - 2):
        for j in range(i + 1, len(sorted_nums) - 1):
            diff = sorted_nums[j] - sorted_nums[i]
            sequence_length = 2
            last_num = sorted_nums[j]
            
            for k in range(j + 1, len(sorted_nums)):
                if sorted_nums[k] == last_num + diff:
                    sequence_length += 1
                    last_num = sorted_nums[k]
            
            max_sequence = max(max_sequence, sequence_length)
    
    return max_sequence

# 알고리즘: 3중 반복문을 통한 등차수열 탐지
# 시간복잡도: O(n³), 공간복잡도: O(1)
```

```python
import numpy as np

def analyze_variance_pattern(numbers):
    """번호 분산도 분석"""
    np_numbers = np.array(numbers)
    
    statistics = {
        'mean': np.mean(np_numbers),
        'std': np.std(np_numbers),
        'variance': np.var(np_numbers),
        'min_gap': np.min(np.diff(np.sort(np_numbers))),
        'max_gap': np.max(np.diff(np.sort(np_numbers)))
    }
    
    # 분산도 분류 (표준편차 기준)
    if statistics['std'] < 10:
        category = '집중형'
    elif statistics['std'] > 15:
        category = '분산형'
    else:
        category = '균형형'
    
    return statistics, category

# 알고리즘: NumPy 통계 함수 활용
# 시간복잡도: O(n log n), 공간복잡도: O(n)
```

### ⚡ 고성능 필터링 엔진

#### FilterOptimizer 클래스 구조
```python
class FilterOptimizer(Generic[T]):
    def __init__(self, process_func, chunk_size=100000, max_workers=None):
        self.process_func = process_func
        self.chunk_size = chunk_size
        self.max_workers = max_workers or min(os.cpu_count(), 8)
    
    def optimize_filter(self, combinations, desc, **kwargs):
        """메인 필터링 로직"""
        if len(combinations) < 10000:
            return self._apply_serial(combinations, desc, **kwargs)
        else:
            return self._apply_parallel(combinations, desc, **kwargs)
```

#### 병렬 처리 알고리즘
```python
def _apply_parallel(self, combinations, desc, **kwargs):
    """병렬 처리를 통한 대용량 데이터 필터링"""
    
    # 1. 데이터 청크 분할
    chunks = [combinations[i:i + self.chunk_size] 
              for i in range(0, len(combinations), self.chunk_size)]
    
    # 2. 프로세스 풀 생성 및 작업 분배
    with ProcessPoolExecutor(max_workers=self.max_workers) as executor:
        futures = {
            executor.submit(self._process_chunk, chunk, **kwargs): chunk 
            for chunk in chunks
        }
        
        # 3. 결과 수집 및 병합
        results = []
        for future in as_completed(futures):
            chunk_result = future.result()
            results.extend(chunk_result)
    
    return results

# 알고리즘: 분할 정복(Divide and Conquer) + 병렬 처리
# 시간복잡도: O(n/p) where p = 프로세스 수
# 메모리 효율성: 청크 단위 처리로 메모리 사용량 최적화
```

#### 분포 분석 알고리즘
```python
def calculate_distribution(data_list, categories):
    """패턴별 분포 계산"""
    total_count = len(data_list)
    distribution = {}
    
    for category in categories:
        count = data_list.count(category)
        percentage = (count / total_count) * 100
        distribution[category] = {
            'count': count,
            'percentage': round(percentage, 2)
        }
    
    return distribution

# 시간복잡도: O(n * m) where n=데이터 크기, m=카테고리 수
# 최적화: Counter 사용 시 O(n)으로 개선 가능
```

#### 구간별 분석 알고리즘
```python
def analyze_range_distribution(numbers, ranges):
    """구간별 번호 분포 분석"""
    range_counts = {range_name: 0 for range_name in ranges.keys()}
    
    for number in numbers:
        for range_name, (start, end) in ranges.items():
            if start <= number <= end:
                range_counts[range_name] += 1
                break
    
    return range_counts

# 예시 구간 정의
ranges = {
    '1-15': (1, 15),
    '16-30': (16, 30), 
    '31-45': (31, 45)
}
```

```python
@staticmethod
def optimize_numpy_operation(func):
    """NumPy 연산 최적화 데코레이터"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        old_settings = np.seterr(all='ignore')  # 경고 무시
        try:
            return func(*args, **kwargs)
        finally:
            np.seterr(**old_settings)
    return wrapper

# NumPy 배열 변환으로 속도 향상
def convert_to_numpy_array(combinations, dtype=np.int8):
    return np.array([list(map(int, comb.split(','))) 
                     for comb in combinations], dtype=dtype)
```

- **진행률 모니터링**: 실시간 처리 상황 확인

### 주요 성능 특징
```python
# 작은 데이터셋 (10,000개 미만): 직렬 처리

max_workers = min(os.cpu_count(), 8)  # 최대 작업자 수

dtype = np.int8  # 1바이트 정수 (1-45 범위)
memory_per_chunk = chunk_size * 6 * 1  # 약 600KB per chunk
```
