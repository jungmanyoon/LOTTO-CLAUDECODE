def _get_optimized_filter_order(self) -> List[Tuple[str, BaseFilter]]:
    """필터 적용 순서 최적화 - 3단계 계층적 구조 적용"""
    # 필터 효율성 업데이트
    self._update_filter_efficiency()
    
    # 필터를 3단계로 구분
    stage1_filters = []  # Stage 1: 높은 제외율, 낮은 계산 비용 (빠른 필터)
    stage2_filters = []  # Stage 2: 중간 제외율, 중간 계산 비용
    stage3_filters = []  # Stage 3: 낮은 제외율, 높은 계산 비용 (느린 필터)
    
    # 각 필터를 단계별로 명시적 분류 (모든 16개 필터 포함)
    stage_mapping = {
        # Stage 1 (기본 필터)
        'sum_range': 1,     # 합계 범위 필터
        'odd_even': 1,      # 홀짝 필터
        'match': 1,         # 번호 일치 필터
        'digit_sum': 1,     # 자릿수 합 필터
        
        # Stage 2 (중간 필터)
        'consecutive': 2,   # 연속 번호 필터
        'max_gap': 2,       # 최대 간격 필터
        'section': 2,       # 구간 필터
        'multiple': 2,      # 배수 필터
        'ten_section': 2,   # 10구간 필터
        'dispersion': 2,    # 분산도 필터
        
        # Stage 3 (고급 필터)
        'fixed_step': 3,    # 고정 간격 필터
        'last_digit': 3,    # 끝자리 숫자 필터
        'average': 3,       # 평균값 필터
        'arithmetic_sequence': 3, # 등차수열 필터
        'geometric_sequence': 3,  # 등비수열 필터
        'prime_composite': 3      # 소수/합성수 필터
    }
    
    # 필터를 단계별로 분류
    for filter_name, filter_obj in self.filters.items():
        efficiency = self.filter_efficiency.get(filter_name, 0.1)
        stage = stage_mapping.get(filter_name, 2)  # 기본값은 Stage 2
        
        if stage == 1:
            stage1_filters.append((filter_name, filter_obj, efficiency))
        elif stage == 2:
            stage2_filters.append((filter_name, filter_obj, efficiency))
        else:
            stage3_filters.append((filter_name, filter_obj, efficiency))
    
    # 각 단계 내에서 효율성 기준 정렬
    stage1_filters.sort(key=lambda x: x[2], reverse=True)
    stage2_filters.sort(key=lambda x: x[2], reverse=True)
    stage3_filters.sort(key=lambda x: x[2], reverse=True)
    
    # 3단계를 순차적으로 결합
    ordered_filters = []
    for stage in [stage1_filters, stage2_filters, stage3_filters]:
        ordered_filters.extend([(name, obj) for name, obj, _ in stage])
    
    # 필터 순서 로깅
    logging.info("\n[필터링 순서 (계층적)]")
    for i, (name, _) in enumerate(ordered_filters, 1):
        logging.info(f"{i:2d}. {name} 필터")
    
    return ordered_filters

def analyze_patterns_with_time_weights(self, round_num: int = None, half_life: int = 10) -> bool:
    """시간 가중치를 적용한 패턴 분석"""
    try:
        logging.info("\n[패턴 분석] 시간 가중치 기반 패턴 분석 시작")
        
        if round_num is None:
            round_num = self.db_manager.lotto_db.get_last_round()
                
        if round_num is None:
            logging.error("분석할 회차 정보를 찾을 수 없습니다.")
            return False
                
        winning_numbers = self.winning_numbers
        if not winning_numbers:
            logging.error("분석할 당첨 번호가 없습니다.")
            return False
                
        total = len(winning_numbers)
        
        # 시간 가중치 계산 (최근 데이터에 높은 가중치)
        weights = []
        for i in range(total):
            # 지수적 감소 가중치 (half_life 회차마다 가중치 절반으로 감소)
            weight = math.exp(-0.693 * i / half_life)
            weights.append(weight)
        
        # 가중치 정규화
        total_weight = sum(weights)
        normalized_weights = [w / total_weight for w in weights]
        
        # 가중치 적용한 패턴 분석 결과 저장
        weighted_patterns = {}
        
        # 1. 홀짝 패턴 가중 분석
        odd_counts = {i: 0 for i in range(7)}
        for i, numbers_str in enumerate(winning_numbers):
            numbers = list(map(int, numbers_str.split(',')))
            odd_count = sum(1 for num in numbers if num % 2 == 1)
            odd_counts[odd_count] += normalized_weights[i]
        weighted_patterns['odd_even'] = odd_counts
        
        # 2. 연속 번호 패턴 가중 분석
        consecutive_counts = {i: 0 for i in range(1, 7)}
        for i, numbers_str in enumerate(winning_numbers):
            numbers = sorted(map(int, numbers_str.split(',')))
            max_consecutive = self._get_max_consecutive(numbers)
            consecutive_counts[max_consecutive] += normalized_weights[i]
        weighted_patterns['consecutive'] = consecutive_counts
        
        # 3-16. 나머지 모든 패턴에 대해 가중치 분석 구현
        # (모든 16개 패턴에 대해 동일한 방식 적용)
        
        # ... (중략) ...
        
        # 패턴 저장
        self.db_manager.save_pattern_analysis(round_num, {
            'weighted_patterns': weighted_patterns,
            'half_life': half_life,
            'analysis_time': datetime.now().isoformat()
        })
        
        self._log_weighted_pattern_analysis(weighted_patterns, half_life)
        return True
        
    except Exception as e:
        logging.error(f"시간 가중치 패턴 분석 중 오류 발생: {str(e)}")
        return False
    
def _get_max_consecutive(self, numbers: List[int]) -> int:
    """가장 긴 연속 번호 개수 계산"""
    max_consecutive = 1
    current_consecutive = 1
    
    for i in range(1, len(numbers)):
        if numbers[i] == numbers[i-1] + 1:
            current_consecutive += 1
            max_consecutive = max(max_consecutive, current_consecutive)
        else:
            current_consecutive = 1
            
    return max_consecutive

def detect_cyclical_patterns(self, historical_data: List[str], max_cycle_length: int = 50) -> Dict[int, float]:
    """순환 패턴 감지 - 자기상관(autocorrelation) 분석 사용
    
    Args:
        historical_data: 당첨 번호 이력
        max_cycle_length: 검색할 최대 주기 길이
        
    Returns:
        Dict[int, float]: 주기별 상관계수 {주기길이: 상관계수}
    """
    try:
        patterns = {}
        
        # 당첨번호를 숫자 배열로 변환
        number_history = []
        for draw in historical_data:
            numbers = [int(n) for n in draw.split(',')]
            number_history.append(numbers)
        
        # 숫자별 출현 이력 생성 (1-45)
        occurrence_history = {}
        for num in range(1, 46):
            # 각 회차별 번호 출현 여부 (0 또는 1)
            occurrence_history[num] = [1 if num in draw else 0 for draw in number_history]
        
        # 각 주기별 자기상관 계산
        for cycle_length in range(5, max_cycle_length + 1):
            # 모든 번호의 해당 주기 상관계수 평균
            correlations = []
            
            for num in range(1, 46):
                series = occurrence_history[num]
                if len(series) <= cycle_length:
                    continue
                    
                # 현재 시퀀스와 cycle_length만큼 이동한 시퀀스 간 상관관계 계산
                original = series[:-cycle_length]
                shifted = series[cycle_length:]
                
                # 자기상관계수 계산 (제로 평균이 아닌 시리즈 처리)
                if np.std(original) > 0 and np.std(shifted) > 0:
                    correlation = np.corrcoef(original, shifted)[0, 1]
                    if not np.isnan(correlation):
                        correlations.append(correlation)
            
            # 모든 번호의 평균 상관계수
            if correlations:
                patterns[cycle_length] = np.mean(correlations)
        
        # 상관계수가 높은 순으로 반환
        CORRELATION_THRESHOLD = 0.2  # 의미 있는 상관관계로 간주할 임계값
        significant_patterns = {
            cycle: corr for cycle, corr in patterns.items() 
            if abs(corr) > CORRELATION_THRESHOLD
        }
        
        return dict(sorted(significant_patterns.items(), key=lambda x: abs(x[1]), reverse=True))
            
    except Exception as e:
        logging.error(f"순환 패턴 감지 중 오류 발생: {str(e)}")
        return {}

def _update_filter_efficiency(self):
    """필터 효율성 동적 업데이트 - 필터 성능 데이터 기반"""
    try:
        # 현재 회차 가져오기
        current_round = self.db_manager.get_last_round()
        
        # 기존 필터링 결과 통계 조회
        stats = self.db_manager.get_filtered_statistics(current_round)
        
        # 필터별 성능 이력 로드
        filter_history = self._load_filter_performance_history(5)  # 최근 5회 데이터
        
        # 성능 기반 초기 효율성 값 설정
        base_efficiency = {
            'sum_range': 0.45,          # 합계 범위
            'odd_even': 0.25,           # 홀짝 
            'consecutive': 0.30,        # 연속 번호
            'max_gap': 0.25,            # 최대 간격
            'section': 0.22,            # 구간
            'geometric_sequence': 0.20, # 등비수열
            'arithmetic_sequence': 0.18, # 등차수열
            'fixed_step': 0.15,         # 고정 간격
            'ten_section': 0.12,        # 10구간
            'last_digit': 0.10,         # 끝자리 숫자
            'average': 0.10,            # 평균값
            'multiple': 0.08,           # 배수
            'match': 0.35,              # 번호 일치
            'prime_composite': 0.15,    # 소수/합성수
            'digit_sum': 0.20,          # 자릿수 합
            'dispersion': 0.18,         # 분산도
        }
        
        # 현재 통계와 이력 데이터를 결합하여 효율성 업데이트
        for filter_name, filter_stats in stats.items():
            if filter_name in self.filters and 'exclude_ratio' in filter_stats:

                self.filter_efficiency[filter_name] = 0.7 * current_ratio + 0.3 * history_ratio
                
        # 필터 성능 초깃값 설정 (통계가 없는 경우)
        for filter_name in self.filters:
            if filter_name not in self.filter_efficiency:
                self.filter_efficiency[filter_name] = base_efficiency.get(filter_name, 0.1)
        
        # 최근 패턴 분석 결과 반영 (PatternManager에서 가져옴)

                if 'odd_even' in patterns and patterns['odd_even']:
                    odd_even_pattern = patterns['odd_even']
                    if any(ratio > 60 for ratio in odd_even_pattern.values()):
                        self.filter_efficiency['odd_even'] *= 1.2
                
                # 합계 범위 편중 시 합계 필터 가중치 증가
                if 'sum_range' in patterns and patterns['sum_range']:
                    sum_ranges = patterns['sum_range']
                    if any(ratio > 30 for ratio in sum_ranges.values()):
                        self.filter_efficiency['sum_range'] *= 1.2
                        
                # 다른 패턴에 대해서도 유사한 로직 적용
                # ... (모든 16개 패턴에 대해 적용) ...
        except Exception as pattern_err:
            logging.warning(f"패턴 분석 결과 적용 중 오류: {str(pattern_err)}")
            
        logging.info("필터 효율성이 업데이트되었습니다.")

class HybridEnsembleModel:

        # LSTM 모델 초기화 (TensorFlow 설치되어 있을 경우)

            'pattern': 0.25
        }
        
        # ML 모델 가중치 추가 (있을 경우)
        if 'random_forest' in self.ml_models:
            self.model_weights['random_forest'] = 0.15

        """모든 모델 학습

            if self.ml_models:
                X, y = self._prepare_ml_training_data(historical_data)
                
                # RandomForest 모델 학습
                if 'random_forest' in self.ml_models:
                    self.ml_models['random_forest'].fit(X, y)
                    logging.info("- RandomForest 모델 학습 완료")
                
                # LSTM 모델 학습
                if 'lstm' in self.ml_models and len(historical_data) >= 20:
                    X_lstm = self._prepare_lstm_data(historical_data)
                    self.ml_models['lstm'].fit(
                        X_lstm, y, 
                        epochs=epochs, 
                        batch_size=32, 
                        verbose=0
                    )
                    logging.info("- LSTM 모델 학습 완료")
            
            logging.info("모든 모델 학습 완료")
            
        except Exception as e:
            logging.error(f"모델 학습 중 오류 발생: {str(e)}")
    
    def predict(self, top_n: int = 10) -> List[List[int]]:
        """확률이 가장 높은 번호 조합 예측
        
        Args:
            top_n: 반환할 상위 조합 수
            
        Returns:
            List[List[int]]: 예측된 번호 조합 목록
        """
        try:
            # 각 모델의 확률 예측값 가져오기
            probability_predictions = {}
            
            # 통계 모델 예측
            for name, model in self.statistical_models.items():
                predictions = model.predict_probabilities()
                probability_predictions[name] = predictions
            
            # ML 모델 예측
            recent_features = self._extract_recent_features()
            
            if 'random_forest' in self.ml_models:
                rf_probs = self.ml_models['random_forest'].predict_proba(recent_features)[:,1]
                probability_predictions['random_forest'] = rf_probs
            
            if 'lstm' in self.ml_models:
                lstm_input = self._prepare_recent_lstm_data()
                lstm_probs = self.ml_models['lstm'].predict(lstm_input)[0]
                probability_predictions['lstm'] = lstm_probs
            
            # 앙상블 확률 계산
            final_probabilities = np.zeros(45)
            
            for model_name, probs in probability_predictions.items():
                if model_name in self.model_weights:
                    weight = self.model_weights[model_name]
                    final_probabilities += probs * weight
            
            # 확률 정규화
            if np.sum(final_probabilities) > 0:
                final_probabilities = final_probabilities / np.sum(final_probabilities)
            
            # 상위 확률 번호 조합 생성
            combinations = self._generate_top_combinations(final_probabilities, top_n)
            
            return combinations
            
        except Exception as e:
            logging.error(f"예측 중 오류 발생: {str(e)}")
            return []
    
    def _prepare_ml_training_data(self, historical_data):
        """머신러닝 모델용 학습 데이터 준비"""

        for i in range(1, len(historical_data)):
            # 이전 회차까지의 데이터로 특성 생성
            features = self._extract_features(historical_data[:i])
            X.append(features)
            
            # 다음 회차의 번호 (타겟 레이블)
            next_draw = historical_data[i]
            label = [1 if j+1 in next_draw else 0 for j in range(45)]
            y.append(label)
        
        return np.array(X), np.array(y)

    def _extract_features(self, data):
        """번호별 특성 추출"""
        features = []
        
        if not data:
            return np.zeros(45 * 4)  # 기본 특성 벡터
        
        # 지난 회차 데이터 바탕으로 특성 생성
        frequency = np.zeros(45)  # 번호별 출현 빈도
        recency = np.ones(45) * len(data)  # 각 번호의 최근 출현 회차
        
        for i, draw in enumerate(data):
            for num in draw:
                frequency[num-1] += 1
                recency[num-1] = min(recency[num-1], len(data) - i)  # 가장 최근 출현
        
        # 빈도 정규화
        frequency = frequency / len(data)
        
        # 최근성 정규화
        recency = recency / len(data)
        
        # 추가 특성 계산
        
        # 1. 홀짝 비율
        odd_count = sum(1 for num in range(1, 46) if num % 2 == 1 and frequency[num-1] > 0)
        even_count = sum(1 for num in range(1, 46) if num % 2 == 0 and frequency[num-1] > 0)
        odd_even_ratio = np.array([odd_count / (odd_count + even_count)] * 45)
        
        # 2. 번호별 예측 점수
        prediction_score = frequency * (1 - recency)  # 높은 빈도, 낮은 최근성 = 높은 점수
        
        # 전체 특성 벡터
        all_features = np.concatenate([
            frequency, 
            recency, 
            odd_even_ratio, 
            prediction_score
        ])
        
        return all_features

    def _generate_top_combinations(self, probabilities, top_n):
        """확률이 가장 높은 번호 조합 생성
        
        Args:
            probabilities: 각 번호의 출현 확률 (1-45)
            top_n: 생성할 조합 수
            
        Returns:
            List[List[int]]: 예측된 조합 목록
        """
        try:
            from itertools import combinations
            
            # 확률 내림차순 정렬 인덱스 (0-based)
            sorted_indices = np.argsort(probabilities)[::-1]
            
            # 높은 확률의 상위 15개 번호 선택
            top_indices = sorted_indices[:15]
            top_numbers = [i+1 for i in top_indices]  # 1-based 번호로 변환
            
            # 15개 번호 중 6개 선택하는 모든 조합 생성
            all_combinations = list(combinations(top_numbers, 6))
            
            # 각 조합의 점수 계산
            combo_scores = []
            for combo in all_combinations:
                # 조합 내 번호들의 확률 합
                score = sum(probabilities[num-1] for num in combo)
                combo_scores.append((list(combo), score))
            
            # 점수 기준 내림차순 정렬
            combo_scores.sort(key=lambda x: x[1], reverse=True)
            
            # 상위 조합 반환
            return [combo for combo, _ in combo_scores[:top_n]]
            
        except Exception as e:
            logging.error(f"조합 생성 중 오류 발생: {str(e)}")
            return []
    
    def update_weights(self, performance_metrics):
        """성능에 기반한 모델 가중치 업데이트
        
        Args:
            performance_metrics: 각 모델의 성능 지표
        """
        try:
            # 모델 가중치 계산
            total_score = sum(performance_metrics.values())
            if total_score <= 0:
                logging.warning("모델 성능 점수가 유효하지 않습니다.")
                return
                
            # 각 모델 가중치 업데이트
            for model_name, score in performance_metrics.items():
                # 최소 가중치는 0.05로 설정
                self.model_weights[model_name] = max(0.05, score / total_score)
            
            # 전체 가중치 합이 1이 되도록 정규화
            total_weight = sum(self.model_weights.values())
            for model_name in self.model_weights:
                self.model_weights[model_name] /= total_weight
                
            logging.info("모델 가중치 업데이트 완료")
            for name, weight in sorted(self.model_weights.items(), key=lambda x: x[1], reverse=True):
                logging.info(f"- {name}: {weight:.4f}")
                
        except Exception as e:
            logging.error(f"가중치 업데이트 중 오류 발생: {str(e)}")

class BaseStatisticalModel:
    """통계 모델 기본 클래스"""

        self.probabilities = np.ones(45) / 45  # 기본 균등 확률
    
    def train(self, historical_data: List[List[int]]):

    def predict_probabilities(self) -> np.ndarray:
        """각 번호의 출현 확률 예측"""
        return self.probabilities

class FrequencyModel(BaseStatisticalModel):
    """출현 빈도 기반 모델"""
    
    def train(self, historical_data: List[List[int]]):
        """빈도 기반 학습"""
        frequency = np.zeros(45)
        total_draws = len(historical_data)
        
        if total_draws == 0:
            self.probabilities = np.ones(45) / 45
            return
            
        # 번호별 출현 빈도 계산
        for draw in historical_data:
            for num in draw:
                frequency[num-1] += 1
        
        # 확률 계산 (출현 빈도 / 총 회차)
        # 0 확률 방지를 위한 라플라스 스무딩 적용
        smooth_freq = frequency + 1
        self.probabilities = smooth_freq / np.sum(smooth_freq)

class GapAnalysisModel(BaseStatisticalModel):
    """출현 간격 분석 모델"""
    
    def train(self, historical_data: List[List[int]]):
        """간격 기반 학습"""
        if not historical_data:
            self.probabilities = np.ones(45) / 45
            return
            
        # 각 번호의 마지막 출현 회차 기록
        last_seen = np.ones(45) * len(historical_data)
        
        # 각 번호의 평균 간격 계산
        gaps = {i+1: [] for i in range(45)}
        current_gap = np.zeros(45)
        
        for i, draw in enumerate(historical_data):
            # 현재 간격 업데이트
            current_gap += 1
            
            # 현재 회차에 출현한 번호 처리
            for num in draw:
                # 간격 기록
                gaps[num].append(current_gap[num-1])
                # 간격 초기화
                current_gap[num-1] = 0
                # 마지막 출현 회차 업데이트
                last_seen[num-1] = len(historical_data) - i
        
        # 평균 간격 계산
        avg_gaps = np.zeros(45)
        for i in range(45):
            if gaps[i+1]:
                avg_gaps[i] = np.mean(gaps[i+1])
            else:
                avg_gaps[i] = len(historical_data)  # 출현하지 않은 경우
        
        # 현재 간격 업데이트 (마지막 회차 이후)
        current_gap = np.zeros(45)
        for i in range(45):
            current_gap[i] = last_seen[i]
            
        # 간격 기반 확률 계산 (현재 간격 / 평균 간격)
        # 간격이 평균에 가까워질수록 확률 증가
        gap_ratios = current_gap / avg_gaps
        gap_probs = 1 - np.exp(-gap_ratios)  # 지수함수 변환
        
        # 0/무한대 방지
        gap_probs = np.clip(gap_probs, 0.001, 0.999)
        
        # 정규화
        self.probabilities = gap_probs / np.sum(gap_probs)

class PatternRecognitionModel(BaseStatisticalModel):
    """패턴 인식 모델"""
    
    def train(self, historical_data: List[List[int]]):
        """패턴 인식 학습"""
        if not historical_data or len(historical_data) < 5:  # 최소 5회 이상 데이터 필요
            self.probabilities = np.ones(45) / 45
            return
            
        # 번호 밸런스 분석
        self.probabilities = self._analyze_number_balance(historical_data)
        
    def _analyze_number_balance(self, historical_data):
        """번호 밸런스(홀짝, 구간, 합계 등) 분석"""
        # 초기 균등 확률
        probabilities = np.ones(45) / 45
        
        # 최근 10회 데이터 추출
        recent_data = historical_data[-10:]
        
        # 1. 홀짝 밸런스
        odd_count = 0
        even_count = 0
        for draw in recent_data:
            for num in draw:
                if num % 2 == 1:
                    odd_count += 1
                else:
                    even_count += 1
        
        total = odd_count + even_count
        odd_ratio = odd_count / total
        even_ratio = even_count / total
        
        # 홀짝 불균형 시 조정
        if abs(odd_ratio - 0.5) > 0.1:  # 불균형 감지
            if odd_ratio < 0.5:  # 홀수가 적게 나온 경우
                # 홀수 번호 확률 증가
                for i in range(45):
                    if (i+1) % 2 == 1:  # 홀수
                        probabilities[i] *= 1.2
            else:  # 짝수가 적게 나온 경우
                # 짝수 번호 확률 증가
                for i in range(45):
                    if (i+1) % 2 == 0:  # 짝수
                        probabilities[i] *= 1.2
        
        # 2. 구간 밸런스
        section_counts = [0, 0, 0, 0, 0]  # 1-9, 10-19, 20-29, 30-39, 40-45
        section_ranges = [(1,9), (10,19), (20,29), (30,39), (40,45)]
        
        for draw in recent_data:
            for num in draw:
                for i, (start, end) in enumerate(section_ranges):
                    if start <= num <= end:
                        section_counts[i] += 1
                        break
        
        section_ratios = [count / sum(section_counts) for count in section_counts]
        expected_ratios = [0.2, 0.2, 0.2, 0.2, 0.12]  # 45개 번호의 구간별 기대 비율
        
        # 구간 불균형 시 조정
        for i, (ratio, expected) in enumerate(zip(section_ratios, expected_ratios)):
            if ratio < expected * 0.7:  # 특정 구간이 기대보다 70% 이하로 나온 경우
                start, end = section_ranges[i]
                # 해당 구간 번호 확률 증가
                for j in range(start-1, end):
                    if j < 45:  # 범위 체크
                        probabilities[j] *= 1.3
        
        # 확률 정규화
        return probabilities / np.sum(probabilities)

def _calculate_dynamic_sum_range(self, winning_numbers: List[str], confidence: float = 0.95) -> Tuple[float, float]:
    """통계적 분석 기반 동적 합계 범위 계산
    
    Args:
        winning_numbers: 당첨 번호 목록
        confidence: 신뢰 구간 (기본값: 0.95)
        
    Returns:
        Tuple[float, float]: (최소 합계, 최대 합계)
    """
    import statistics
    import math
    
    # 합계 리스트 생성
    sums = []
    for numbers_str in winning_numbers:
        numbers = list(map(int, numbers_str.split(',')))
        sums.append(sum(numbers))
    
    # 통계 계산
    mean = statistics.mean(sums)
    std = statistics.stdev(sums) if len(sums) > 1 else 10
    
    # 신뢰 구간 계산 (z-값 사용)
    z_value = 1.96  # 95% 신뢰 구간의 z-값
    if confidence == 0.99:
        z_value = 2.576
    elif confidence == 0.90:
        z_value = 1.645
    
    margin = z_value * std
    lower_bound = max(21, math.floor(mean - margin))  # 최소 21 (모두 1일 때보다 큰 값)
    upper_bound = min(270, math.ceil(mean + margin))  # 최대 270 (모두 45일 때보다 작은 값)
    
    return lower_bound, upper_bound

def update_dynamic_criteria(self, winning_numbers: List[str]) -> None:
    """당첨 번호 분석 기반 기준값 업데이트
    
    Args:
        winning_numbers: 당첨 번호 목록
    """
    try:
        min_sum, max_sum = self._calculate_dynamic_sum_range(winning_numbers)
        
        # 기준값 업데이트
        self.criteria['min_sum'] = min_sum
        self.criteria['max_sum'] = max_sum
        
        logging.info(f"합계 범위 필터 기준 동적 업데이트: {min_sum}~{max_sum}")
    except Exception as e:
        logging.error(f"합계 범위 동적 업데이트 중 오류 발생: {str(e)}")

            latest_round = db_manager.lotto_db.get_last_round()

            cyclical_patterns = pattern_manager.detect_cyclical_patterns(winning_numbers)
            if cyclical_patterns:

5.1 계층적 필터링 시스템

3단계 계층적 필터링:

Stage 1 (기본 필터): 합계 범위, 홀짝, 번호 일치, 자릿수 합
Stage 2 (중간 필터): 연속 번호, 최대 간격, 구간, 배수, 10구간, 분산도
Stage 3 (고급 필터): 고정 간격, 끝자리 숫자, 평균값, 등차수열, 등비수열, 소수/합성수


동적 필터 효율성: 필터 성능 데이터와 최근 패턴 분석 결과를 기반으로 필터 가중치를 자동 조정하여 가장 효율적인 필터 순서 결정

5.2 시간 가중치 패턴 분석

지수 감소 가중치: 최근 당첨 번호에 더 높은 가중치를 부여하는 분석 시스템
지정된 반감기: 설정된 회차마다 가중치가 절반으로 감소 (기본값: 10회차)
모든 16개 패턴을 시간 가중치로 분석: 홀짝, 연속 번호, 합계 등 모든 패턴에 적용

5.3 순환 패턴 감지

자기상관 분석: 각 번호의 출현 패턴에서 주기성을 찾기 위한 자기상관 계산
유의미한 주기 식별: 설정된 임계값 이상의 상관계수를 가진 주기만 선택
최대 50회차 주기 탐색: 단기부터 장기까지 다양한 주기성 검색

5.4 하이브리드 앙상블 예측 모델

통계 모델 결합:

빈도 모델: 출현 빈도에 기반한 예측
간격 모델: 출현 간격 패턴에 기반한 예측
패턴 모델: 번호 밸런스 및 패턴 인식 기반 예측


머신러닝 모델 통합 (선택적):

RandomForest 모델: 다양한 특성 기반 예측
LSTM 모델: 시계열 패턴 기반 예측


동적 모델 가중치: 각 모델의 예측 성능에 따라 가중치 자동 조정

5.5 동적 필터 기준값 최적화

통계적 기반 임계값: 신뢰 구간 기반 동적 범위 설정 (예: 합계 범위)
이전 필터링 결과 반영: 이전 필터링 효과를 분석하여 기준값 최적화

5.6 모든 필터 통합 적용

모든 16개 필터가 최적의 순서로 계층적으로, 그리고 효율적으로 적용됨
필터링 결과가 실시간으로 모니터링되고 시각화됨

5.7 성능 최적화

병렬 처리: 대규모 조합 처리를 위한 멀티프로세싱
메모리 최적화: 배치 처리 및 메모리 사용량 제한
계산 효율성: NumPy 벡터화 연산 및 비트 연산으로 성능 향상

이 개선된 알고리즘은 기존 코드의 구조와 모든 16개 필터를 유지하면서, 새로운 분석 방법과 예측 모델을 통합하여 예측 정확도를 향상시킵니다.
